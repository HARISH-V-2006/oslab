#include <iostream>
#include <vector>
using namespace std;

int main() {
    string dirName, f;
    vector<string> files;
    int ch;

    cout << "Enter directory name: ";
    cin >> dirName;

    while (true) {
        cout << "\n1.Create 2.Delete 3.Search 4.Display 5.Exit\nChoice: ";
        cin >> ch;

        if (ch == 1) {
            if (files.size() < 10) {
                cout << "Enter file name: "; cin >> f;
                files.push_back(f);
            } else cout << "Directory Full!\n";
        } 
        else if (ch == 2) {
            cout << "Enter file to delete: "; cin >> f;
            bool ok = false;
            for (int i = 0; i < files.size(); i++)
                if (files[i] == f) { files[i] = files.back(); files.pop_back(); cout << f << " deleted\n"; ok = true; break; }
            if (!ok) cout << f << " not found\n";
        } 
        else if (ch == 3) {
            cout << "Enter file to search: "; cin >> f;
            bool ok = false;
            for (auto &x : files) if (x == f) { cout << f << " found\n"; ok = true; break; }
            if (!ok) cout << f << " not found\n";
        } 
        else if (ch == 4) {
            if (files.empty()) cout << "Directory Empty\n";
            else { cout << "Files:\n"; for (auto &x : files) cout << x << "\n"; }
        } 
        else if (ch == 5) break;
        else cout << "Invalid choice!\n";
    }
}
