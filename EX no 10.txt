FCFS

#include <iostream>
#include <cstdlib> // for abs()
using namespace std;

int main() {
    int RQ[100], n, TotalHeadMoment = 0, initial;

    cout << "Enter the number of Requests\n";
    cin >> n;

    cout << "Enter the Requests sequence\n";
    for (int i = 0; i < n; i++)
        cin >> RQ[i];

    cout << "Enter initial head position\n";
    cin >> initial;

    // logic for FCFS disk scheduling
    for (int i = 0; i < n; i++) {
        TotalHeadMoment += abs(RQ[i] - initial);
        initial = RQ[i];
    }

    cout << "Total head movement is " << TotalHeadMoment << endl;
    return 0;
}


SSTF

#include <iostream>
#include <cstdlib> // for abs()
using namespace std;

int main() {
    int RQ[100], n, TotalHeadMoment = 0, initial, count = 0;

    cout << "Enter the number of Requests\n";
    cin >> n;

    cout << "Enter the Requests sequence\n";
    for (int i = 0; i < n; i++)
        cin >> RQ[i];

    cout << "Enter initial head position\n";
    cin >> initial;

    // logic for SSTF disk scheduling
    while (count != n) {
        int min = 1000, d, index = -1;

        for (int i = 0; i < n; i++) {
            d = abs(RQ[i] - initial);
            if (min > d) {
                min = d;
                index = i;
            }
        }

        TotalHeadMoment += min;
        initial = RQ[index];
        RQ[index] = 1000; // mark as visited
        count++;
    }

    cout << "Total head movement is " << TotalHeadMoment << endl;
    return 0;
}
